23def bfs(graph: List[List[Edge]], parent: List[int], s: int, t: int) -> bool:
24 n = len(graph)

40def augment(graph: List[List[Edge]], parent: List[int], s: int, t: int) -> int:
41 n = len(graph)
42 min_flow = float('inf')
43 v = t
44 while v != s:
45 u = parent[v]
46 for e in graph[u]:
47 if e.u == v:
48 min_flow = min(min_flow, e.capacity - e.flow)
49 break
50 v = u
51 v = t
52 while v != s:
53 u = parent[v]
54 for e in graph[u]:
55 if e.u == v:
56 e.flow += min_flow
57 break
58 for e in graph[v]:
59 if e.u == u:
60 e.flow -= min_flow
61 break
62 v = u
63 return min_flow

65def ford_fulkerson(graph: List[List[Edge]], s: int, t: int) -> int:
66 n = len(graph)
67 parent = [0] * n
68 max_flow = 0
69 while bfs(graph, parent, s, t):
70 flow = augment(graph, parent, s, t)
71 max_flow += flow
72 return max_flow

74def read_graph_from_file(filepath: str) -> Tuple[int, int, int, int, List[List[Edge]]]:
75 with open(filepath) as f:
