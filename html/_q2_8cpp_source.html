47double cost(vector<pair<double, double>>& points, int i, int j, double c) {
48 double sum_x = 0, sum_y = 0, sum_xy = 0, sum_xx = 0;
49 for (int k = i; k <= j; k++) {
50 sum_x += points[k].first;
51 sum_y += points[k].second;
52 sum_xy += points[k].first * points[k].second;
53 sum_xx += points[k].first * points[k].first;
54 }
55 double n = j - i + 1;
56 double denominator = n * sum_xx - sum_x * sum_x;
57 if (denominator == 0) {
58 return INT_MAX;
59 }
60 double slope = (n * sum_xy - sum_x * sum_y) / denominator;
61 double intercept = (sum_y - slope * sum_x) / n;
62 double res = 0;
63 for (int k = i; k <= j; k++) {
64 double error = points[k].second - (slope * points[k].first + intercept);
65 res += error * error;
66 }
67 return res + c;
68}

78pair<vector<pair<int, int>>, double> segmented_least_squares(vector<pair<double, double>>& points, double c) {
79 int n = points.size();
80 bool same_line = true;
81 for (int i = 1; i < n; i++) {
82 if ((points[i].second - points[0].second) * (points[i - 1].first - points[0].first) != (points[i - 1].second - points[0].second) * (points[i].first - points[0].first)) {
83 same_line = false;
84 break;
85 }
86 }
87 if (same_line) {
88 return {{}, 0};
89 }
90 // dp[j] = min(ei,j + c + dp[i - 1]) for all i <= j
91 vector<double> dp(n);
92 vector<int> prev(n);
93 for (int i = 0; i < n; i++) {
94 dp[i] = cost(points, 0, i, c);
95 prev[i] = -1;
96 }
97 for (int i = 1; i < n; i++) {
98 for (int j = i; j < n; j++) {
99 double curr_cost = cost(points, i, j, c) + dp[i - 1];
100 if (curr_cost < dp[j]) {
101 dp[j] = curr_cost;
102 prev[j] = i - 1;
103 }
104 }
105 }
106 vector<pair<int, int>> segments;
107 int curr = n - 1;
108 while (curr != -1) {
109 segments.pb({prev[curr] + 1, curr});
110 curr = prev[curr];
111 }
112 reverse(segments.begin(), segments.end());
113 return {segments, dp[n - 1]};
114}

123int count_partitions(vector<pair<int, int>>& segments, int n) {
124 int cnt = 0;
125 int last = -1;
126 for (auto& p : segments) {
127 if (p.first > last) {
128 cnt++;
129 }
130 last = max(last, p.second);
131 }
132 if (last < n - 1) {
133 cnt++;
134 }
135 return cnt;
