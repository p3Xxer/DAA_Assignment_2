77{
78 int n = graph.size();
79 vector<bool> visited(n, false);
80 queue<int> q;
81 q.push(s);
82 visited[s] = true;
83 parent[s] = -1;
84 while (!q.empty())
85 {
86 int u = q.front();
87 q.pop();
88 visited[u] = true;
89 for (auto &e : graph[u])

91 int v = e.u;
